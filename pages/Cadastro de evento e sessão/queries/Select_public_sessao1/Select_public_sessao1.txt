SELECT
  *
FROM
  public.sessao
WHERE
  "nomesessao" ilike '%{{tblSessao.searchText}}%' 
	AND "evento_id" = {{tblEvento.selectedRow.id}}
ORDER BY
  "{{tblSessao.sortOrder.column || 'id'}}" {{tblSessao.sortOrder.order !== "desc" ? "" : "DESC"}}
LIMIT
  {{tblSessao.pageSize}}
OFFSET
  {{tblSessao.pageOffset}}
